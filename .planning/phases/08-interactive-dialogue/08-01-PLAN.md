```typescript
export const simulationApi = {
  interview(params: {
    simulation_id: string
    agent_id: string
    prompt: string
    platform?: string
    timeout?: number
  }) {
    return apiRequest('/simulation/interview', {
      method: 'POST',
      body: JSON.stringify(params),
      schema: InterviewResponseSchema,
    })
  },

  getProfiles(simId: string, platform?: string) {
    const query = platform ? `?platform=${platform}` : ''
    return apiRequest(`/simulation/${simId}/profiles${query}`, {
      schema: ProfilesResponseSchema,
    })
  },
}
```

   - Return: `{ messages, target, send: sendMutation.mutate, switchAgent, isPending: sendMutation.isPending }`
